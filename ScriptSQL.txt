create database hstestdb4;

use hstestdb4;

CREATE TABLE HTML (
	uuid CHAR(36),
	content TEXT,
        PRIMARY KEY (uuid)
);

CREATE TABLE XML (
	uuid CHAR(36),
	content TEXT,
        PRIMARY KEY (uuid)
);

CREATE TABLE XSD (
	uuid CHAR(36),
	content TEXT,
        PRIMARY KEY (uuid)
);

CREATE TABLE XSLT (
	uuid CHAR(36),
	xsd CHAR(36),
	content TEXT,
        PRIMARY KEY (uuid)
);

CREATE USER 'hsdb'@'%' IDENTIFIED BY 'hsdbpass';

GRANT ALL PRIVILEGES ON *.* TO 'hsdb'@'%' WITH GRANT OPTION;
